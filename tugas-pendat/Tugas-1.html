def dt1(i,j):
  P = 9
  Q = 0
  for col in ["Manufacturer", "Model", "Prod. year", "Category", "Fuel type", "Gear box type", "Doors", "Wheel", "Color"]:
    if data[col][i] == data[col][j]:
      Q += 1
  return (P-Q)/P

print(dt1(1,2))
print(dt1(1,3))
print(dt1(1,4))

def dt2(i,j,h):
  result = 0
  for col in ["Price", "Cylinders", "Airbags"]:
    result = math.pow(abs(data[col][i] - data[col][j]), h)
  return result**(1/h)

print(dt2(1,2,1))
